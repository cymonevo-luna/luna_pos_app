## Jenkins host commands (to complete checklist)

```bash
cd luna_pos_app
git fetch origin bug/POS-100-1 && git checkout bug/POS-100-1
scripts/ensure-flutter-test-env.sh
eval "$(scripts/ensure-flutter-test-env.sh --export)"

# Step 1
test -f .env || cp .env.example .env
make ci

# Step 2
curl -sS https://pos-api.cymonevo.com/healthz

# Step 3
cat > .env <<'EOF'
API_BASE_URL=https://pos-api.cymonevo.com
SENTRY_DSN=
APP_ENV=staging
EOF

# Step 4
SKIP_CODEGEN=1 scripts/build-apk.sh release
unzip -p build/app/outputs/flutter-apk/app-release.apk assets/.env | grep API_BASE_URL

# Step 5–6
DEVICE="$(scripts/start-shared-emulator.sh | tail -n1)"
adb -s "$DEVICE" install -r build/app/outputs/flutter-apk/app-release.apk
adb -s "$DEVICE" shell am start -n com.example.flutter_template/.MainActivity
```
